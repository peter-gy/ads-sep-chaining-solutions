/**
 * Diese soll genau dann true liefern, wenn
 *  - a und b beide leer sind, oder
 *  - mehr als 90 % der Elemente von a auch in b enthalten sind und
 *    mehr als 90 % der Elemente von b auch in a enthalten sind.
 * Erlaubt ist nur das Verwenden der Methode ADS_set::count.
 */
friend bool z(const ADS_set& a, const ADS_set& b) {
    if (a.curr_size == 0 && b.curr_size == 0) return true;
    size_t num_common_elements {0};
    const auto &small_set {a.curr_size < b.curr_size ? a : b};
    const auto &big_set {a.curr_size > b.curr_size ? a : b};
    for (size_t i {0}; i < big_set.capacity; ++i) {
        auto *head {&big_set.table[i]};
        if (head->is_free) continue;
        while(head) {
            if (small_set.count(head->key)) ++num_common_elements;
            head = head->next;
        }
    }
    return 1.0*num_common_elements > 0.9*big_set.curr_size;
}

/**
 * Diese soll genau dann true liefern, wenn in der durch einen Iterator gelieferten Reihenfolge der Schlüsselwerte
 * - a unmittelbar hinter b, oder
 * - b unmittelbar hinter a liegt.
 * Erlaubt ist nur das Verwenden von std::equal_to für den Vergleich von Schlüsselwerten.
 */
bool y(const key_type& a, const key_type& b) const {
    if (curr_size < 2) return false;
    Node *prev {nullptr};
    for (size_type i {0}; i < capacity; ++i) {
        Node *curr {&table[i]};
        if (curr->is_free) continue;
        while (curr) {
            if (!prev) {
                prev = curr;
                curr = curr->next;
                continue;
          
            bool a_before_b {key_equal{}(prev->key, a) && key_equal{}(curr->key, b)};
            if (a_before_b) return tr
            bool b_before_a {key_equal{}(prev->key, b) && key_equal{}(curr->key, a)};
            if (b_before_a) return tr
            prev = curr;
            curr = curr->next;
        }
    }
    return false;
}

/**
 * Diese soll den Abstand der Werte a und b in Bezug auf die von einem Iterator gelieferten Reihenfolge der
 * Schlüsselwerte retournieren (als Absolutbetrag, also nicht negativ).
 * Wenn zumindest einer der beiden Werte nicht im ADS_set enthalten ist, dann liefert x
 * eine Exception vom Typ std::runtime_error.
 */
size_t x(const key_type& a, const key_type& b) const {
    if (key_equal{}(a, b)) return 0;
    bool first_reached {false};
    size_t distance {0};
    for (size_type i {0}; i < capacity; ++i) {
        Node *curr {&table[i]};
        if (curr->is_free) continue;
        while (curr) {
            if (key_equal{}(curr->key, a) || key_equal{}(curr->key, b)) {
                // will be triggered only on the second time the if is entered
                if (first_reached) return distance;
                first_reached = true;
            }
            if (first_reached) ++distance;
            curr = curr->next;
        }
    }
    throw std::runtime_error("One of the keys are not in the set");
}

/**
 * Diese liefert als Ergebnis die Anzahl der Elemente im Schnitt der Mengen a und b dividiert durch
 * die Anzahl der Elemente in der Vereinigung der Mengen a und b. Wenn a und b beide leer sind,
 * dann ist die Anzahl der Elemente in der Vereinigung 0 und die Division kann nicht durchgeführt werden.
 * In diesem Fall ist der Wert 1 zu retournieren.
 */
friend double z(const ADS_set &a, const ADS_set &b) {
    if (a.curr_size == 0 && b.curr_size == 0) return 1;
    size_t num_common {0};
    const ADS_set &bigger_set = a.curr_size > b.curr_size ? a : b;
    const ADS_set &smaller_set = a.curr_size > b.curr_size ? b : a;
    for (size_t i{0}; i < bigger_set.capacity; ++i) {
        Node *curr {&bigger_set.table[i]};
        if (curr->is_free) continue;
        while (curr) {
            num_common += smaller_set.count(curr->key);
            curr = curr->next;
        }
    }
    size_t union_count {a.curr_size + b.curr_size - num_common};
    return (1.0*num_common) / (1.0*union_count);
}

/**
 * Diese soll die Länge der längsten, zusammenhängenden, aufsteigend sortierten Teilfolge in der
 * durch einen Iterator gelieferten Reihenfolge der Schlüsselwerte liefern.
 */
size_t y() const {
    size_t length {0};
    size_t max_length {0};
    Node *prev {nullptr};
    for (size_type i {0}; i < capacity; ++i) {
        Node *curr {&table[i]};
        if (curr->is_free) continue;
        while (curr) {
            bool end_of_subsequence {prev && key_compare{}(curr->key, prev->key)};
            if (end_of_subsequence) {
                if (length > max_length) max_length = length;
                length = 0;
            }
            ++length;
            prev = curr;
            curr = curr->next;
        }
    }
    return length > max_length ? length : max_length;
}

/**
 * Diese soll das kleinste lokale Maximum in der durch einen Iterator gelieferten Reihenfolge 
 * der Schlüsselwerte liefern. Wenn es kein lokales Maximum gibt, soll eine Exception 
 * vom Typ std::runtime_error ausgelöst werden. 
 * 
 * Zur Vereinfachung wird für diese Aufgabe festgelegt: Ein lokales Maximum ist ein Wert, 
 * der größer ist als sein Vorgänger und größer als sein Nachfolger. Ein lokales Maximum muss daher 
 * einen Vorgänger und einen Nachfolger haben.
 */
key_type x() const {
    if (curr_size < 3) throw std::runtime_error("Set size smaller than 3");
    Node *prev {nullptr}, *mid {nullptr}, *min {nullptr};
    size_type count {0};
    for (size_type i {0}; i < capacity; ++i) {
        Node *curr {&table[i]};
        if (curr->is_free) continue;
        while (curr) {
            // setting prev, mid and next in the first 2 iterations
            if (count < 2) {
                if (count == 0) prev = curr;
                else if (count == 1) mid = curr;
                ++count;
                curr = curr->next;
                continue;
            }
            bool mid_greater_than_prev {key_compare{}(prev->key, mid->key)};
            bool mid_greater_than_next {key_compare{}(curr->key, mid->key)};
            bool local_maxima_found {mid_greater_than_prev && mid_greater_than_next};
            if (local_maxima_found) {
                bool new_min_found {!min || key_compare{}(mid->key, min->key)};
                if (new_min_found) min = mid;
            }
            prev = mid;
            mid = curr;
            curr = curr->next;
        }
    }
    if (!min) throw std::runtime_error("No local maxima was found");
    return min->key;
}

/**
 * Diese soll einen Iterator im Modus „speziell“ liefern. Wenn a oder b nicht im ADS_set enthalten sind,
 * oder wenn a in Iteratorreihenfolge nicht vor b liegt, dann gilt x() == end().
 * Andernfalls liefert x() einen Iterator, der auf a zeigt. Der von x() gelieferte Iterator soll
 * beim Inkrementieren in weiterer Folge alle Werte in Iteratorreihenfolge liefern,
 * allerdings nur bis (exklusive) b. Sobald b erreicht wird, wird end() geliefert.
 */
const_iterator x(const key_type &a, const key_type &b) const {
    if (!count(a) || !count(b)) return end();
    const_iterator it {this, b};
    while (it != end() && !key_equal{}(*it, a)) {
        if (key_equal{}(*it, b)) return end(); // upper bound reached sooner than lower bound
        ++it;
    }
    return it;
}

// ...

bool special {false};
value_type stop;

// ...

explicit Iterator(const ADS_set *set = nullptr, size_type slot = 0, Node *pos = nullptr) : set{set}, slot{slot}, pos{pos} {
    if (set) {
        skip_free_slots();
        bool slot_was_skipped {this->slot != slot};
        if (slot_was_skipped)
            this->pos = &set->table[this->slot];
    }
}

// separate constructor necessary, as no default value can be given to value_type stop
explicit Iterator(const ADS_set *set, value_type stop) : Iterator(set, 0, &set->table[0]) {
    this->stop = stop;
    this->special = true;
}

// ...

Iterator &prefix_increment_special() {
    // increment then check if the stop value is reached
    prefix_increment_normal();
    if (key_equal{}(pos->key, stop)) {
        // set this to end
        slot = set->capacity;
        pos = &set->table[slot];
    }
    return *this;
}

/**
 * Diese soll den ersten und letzten Wert der längsten, zusammenhängenden, aufsteigend sortierten Teilfolge
 * in der durch einen Iterator gelieferten Reihenfolge der Werte liefern.
 * Enthält das ADS_set mehrere solche Teilfolgen mit gleicher Länge, so ist der erste und letzte Wert der
 * ersten Teilfolge zu liefern. Wenn keine Werte und somit auch keine nichtleeren Teilfolgen vorhanden sind,
 * ist eine Exception vom Typ std::runtime_error auszulösen.
 */
std::pair<key_type,key_type> x() const {
    if (curr_size == 0) throw std::runtime_error("Set is empty");
    Node *prev {nullptr};
    size_type length {0}, max_length {0}; // used to track current and max subsequence length
    std::pair<key_type,key_type> pair; // always holds the current boundaries of the subsequence with max length
    Node *first {nullptr}; // used to track the first node of a new subsequence
    for (size_type i {0}; i < capacity; ++i) {
        Node *curr {&table[i]};
        if (curr->is_free) continue;
        while (curr) {
            if (!first) first = curr; // very first element of chain
            bool subsequence_ended {prev && !key_compare{}(prev->key, curr->key)};
            if (subsequence_ended) {
                // first node of the ended subsequence is *first, the last is *prev
                if (max_length < length) {
                    max_length = length;
                    pair.first = first->key;
                    pair.second = prev->key;
                }
                length = 0;
                first = curr;
            }
            ++length;
            prev = curr;
            curr = curr->next;
        }
    }
    if (max_length < length) {
        pair.first = first->key;
        pair.second = prev->key;
    }
    return pair;
}

/**
 * Diese soll den Abstand zwischen dem kleinsten und dem zweitkleinsten gespeicherten Wert in Bezug
 * auf die von einem Iterator gelieferten Reihenfolge der Elemente retournieren
 * (als Absolutbetrag, also nicht negativ). Wenn weniger als zwei Werte im ADS_set enthalten sind,
 * dann liefert x eine Exception vom Typ std::runtime_error
 * (gegebenenfalls müssen Sie zusätzlich <stdexcept> inkludieren).
 */
size_t x() const {
    if (curr_size < 2) throw std::runtime_error("Less than 2 elements in set");
    // obtain absolute min
    Node *min_1 {nullptr};
    size_type index {0}, min_1_index {0};
    for (size_type i {0}; i < capacity; ++i) {
        Node *curr {&table[i]};
        if (curr->is_free) continue;
        while (curr) {
            if (!min_1 || key_compare{}(curr->key, min_1->key)) {
                min_1 = curr;
                min_1_index = index;
            }
            ++index;
            curr = curr->next;
        }
    }

    // obtain second min
    Node *min_2 {nullptr};
    size_type min_2_index {0};
    index = 0;
    for (size_type i {0}; i < capacity; ++i) {
        Node *curr {&table[i]};
        if (curr->is_free) continue;
        while (curr) {
            bool first_min_2_found {!min_2 && key_compare{}(min_1->key, curr->key)};
            if (first_min_2_found) {
                min_2 = curr;
                min_2_index = index;
            }
            bool new_min_2_found = key_compare{}(min_1->key, curr->key) && key_compare{}(curr->key, min_2->key);
            if (new_min_2_found) {
                min_2 = curr;
                min_2_index = index;
            }
            ++index;
            curr = curr->next;
        }
    }
    size_type lower_index = min_1_index > min_2_index ? min_2_index : min_1_index;
    size_type upper_index = lower_index == min_1_index ? min_2_index : min_1_index;
    return upper_index - lower_index;
}

const_iterator w() const {
    if (curr_size == 0) return end();
    Node *min {nullptr};
    size_type min_slot_index {0};
    for (size_type i {0}; i < capacity; ++i) {
        Node *curr {&table[i]};
        if (curr->is_free) continue;
        while (curr) {
            if (!min || key_compare{}(curr->key, min->key)) {
                min = curr;
                min_slot_index = i;
            }
            curr = curr->next;
        }
    }
    return const_iterator{this, min_slot_index, min, true};
}
